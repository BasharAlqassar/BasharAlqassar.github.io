public class Goblin : MonoBehaviour
            {
                [SerializeField] private float m_spin_per_update = 0.5f;
                private PlayerController m_player = null; 
            
                void Start()
                {
                    m_player = FindAnyObjectByType<PlayerController>();
                    m_player.PlayerDied.AddListener(GoblinDespawn);
                    LevelManager.Instance.FocusScreenCalled.AddListener(GoblinDespawn);
                    LevelManager.Instance.CollectableTempCollected.AddListener(GoblinSpawn);
            
                    // Spawn in to avoid freeze on first load
                    GoblinSpawn();
            
                }
            
                void FixedUpdate()
                {
                    transform.Rotate(new Vector3(0f, m_spin_per_update, 0f));
                }
            
                void GoblinSpawn()
                {
                    gameObject.SetActive(true);
                }
            
                void GoblinDespawn()
                {
                    gameObject.SetActive(false);
                }
            
            }

Modular Design
Every single component of the game is modular and carefully crafted to be easily changed on a moment's notice. Colorable objects are made to have their type swapped on the fly with a dedicated and secure pipeline to maintain seamlessness during gameplay.
Rapid Iteration
Engine-side systems are implemented with a focus on being able to quickly prototype an idea, test it and gain immediate feedback for iteration. This allowed for rapid developer-side playtesting during the design phase to ensure quality, to-the-point playtesting with real players.
